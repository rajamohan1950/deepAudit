Integrate in 30 Seconds

The simplest audit API ever built. One endpoint. One line of JSON.

# One line. That's it.
curl -X POST https://your-host/api/v1/quick-audit \
  -H "Content-Type: application/json" \
  -d '{"github_url": "https://github.com/your-org/your-repo"}'
# Response:
{
  "audit_id": "b3d7b523-6589-4616-8350-d143bbf01f08",
  "status": "pending",
  "message": "Audit queued! Analyzing across 40 categories.",
  "poll_url": "/api/v1/audits/b3d7b523.../progress",
  "api_key": "da_x9dlcH0D..."  // save this for future calls
}
import requests

# Start an audit
resp = requests.post("https://your-host/api/v1/quick-audit", json={
    "github_url": "https://github.com/your-org/your-repo"
})
audit = resp.json()
print(f"Audit {audit['audit_id']} started!")

# Poll for results
import time
while True:
    progress = requests.get(
        f"https://your-host{audit['progress_url']}",
        headers={"X-API-Key": audit["api_key"]}
    ).json()
    print(f"Phase {progress['current_phase']}/10 — {progress['signals_so_far']} signals")
    if progress["status"] == "completed": break
    time.sleep(30)
// Start an audit in one fetch call
const resp = await fetch("https://your-host/api/v1/quick-audit", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({
    github_url: "https://github.com/your-org/your-repo"
  })
});
const { audit_id, api_key, poll_url } = await resp.json();

// Poll for results
const check = async () => {
  const r = await fetch(`https://your-host${poll_url}`, {
    headers: { "X-API-Key": api_key }
  });
  return r.json();
};
