# Start an audit — that's it!
curl -X POST https://deepaudit-api.onrender.com/api/v1/quick-audit \
  -H "Content-Type: application/json" \
  -d '{"github_url": "https://github.com/your-org/your-repo"}'

# Response includes audit_id + dashboard URL
# View results at: /audit?id=AUDIT_ID&key=API_KEY
import requests

r = requests.post(
    "https://deepaudit-api.onrender.com/api/v1/quick-audit",
    json={"github_url": "https://github.com/your-org/your-repo"}
)
data = r.json()
print(f"Dashboard: /audit?id={data['audit_id']}&key={data['api_key']}")
const res = await fetch(
  'https://deepaudit-api.onrender.com/api/v1/quick-audit',
  {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      github_url: 'https://github.com/your-org/your-repo'
    })
  }
);
const { audit_id, api_key } = await res.json();
window.open(`/audit?id=${audit_id}&key=${api_key}`);
# .github/workflows/deepaudit.yml
name: DeepAudit
on: [push]
jobs:
  audit:
    runs-on: ubuntu-latest
    steps:
      - name: Run DeepAudit
        run: |
          curl -X POST https://deepaudit-api.onrender.com/api/v1/quick-audit \
            -H "Content-Type: application/json" \
            -d '{"github_url": "${{ github.server_url }}/${{ github.repository }}"}'
